# availability_ai
demo AI implementation for the project management tool availability


# Activate your virtual environment first

source availability/bin/activate
 
# Update all LangChain packages

pip install --upgrade langchain langchain-core langchain-google-genai langgraph
 
# If that doesn't work, try uninstalling and reinstalling

pip uninstall langchain langchain-core langchain-google-genai langgraph -y

pip install langchain langchain-core langchain-google-genai langgraph
